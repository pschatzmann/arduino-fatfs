195#if FF_IO_USE_IOCTL == 1
196 DRESULT disk_ioctl(BYTE pdrv, ioctl_cmd_t cmd, void *buff) override {
197 (void)pdrv;
198 DRESULT res = RES_ERROR;
199
200 if (status & STA_NOINIT) return RES_NOTRDY;
201
202 switch (cmd) {
203 /* Make sure that no pending write process */
204 case CTRL_SYNC:
205 res = RES_OK;
206 break;
207
208 /* Get number of sectors on the disk (DWORD) */
209 case GET_SECTOR_COUNT:
210 BSP_SD_GetCardInfo(&cardInfo);
211 *(DWORD *)buff = cardInfo.LogBlockNbr;
212 res = RES_OK;
213 break;
214
215 /* Get R/W sector size (WORD) */
216 case GET_SECTOR_SIZE:
217 BSP_SD_GetCardInfo(&cardInfo);
218 *(WORD *)buff = cardInfo.LogBlockSize;
219 res = RES_OK;
220 break;
221
222 /* Get erase block size in unit of sector (DWORD) */
223 case GET_BLOCK_SIZE:
224 BSP_SD_GetCardInfo(&cardInfo);
225 *(DWORD *)buff = cardInfo.LogBlockSize / SD_DEFAULT_BLOCK_SIZE;
226 res = RES_OK;
227 break;
228
229 default:
230 res = RES_PARERR;
231 }
232
233 return res;
234 }
