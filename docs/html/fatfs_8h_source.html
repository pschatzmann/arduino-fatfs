198#if FF_FS_MINIMIZE == 0
199 DWORD fre_clust, fre_sect;

201
202 // Get volume information and free clusters of default drive
203 auto res = fs->f_getfree("", &fre_clust, &fatfs);
204 if (res) return 0;
205
206#if FF_MAX_SS != FF_MIN_SS

214#else
