193#if FF_FS_MINIMIZE == 0
194 DWORD fre_clust, fre_sect;

196
197 // Get volume information and free clusters of default drive
198 auto res = fs->f_getfree("", &fre_clust, &fatfs);
199 if (res) return 0;

201#if FF_MAX_SS != FF_MIN_SS
