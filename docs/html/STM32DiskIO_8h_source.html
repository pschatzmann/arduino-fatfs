142 #if _USE_IOCTL == 1
143  DRESULT disk_ioctl(BYTE lun, BYTE cmd, void *buff) {
144  (void)lun;
145  DRESULT res = RES_ERROR;
146  BSP_SD_CardInfo CardInfo;
147 
148  if (Stat & STA_NOINIT) return RES_NOTRDY;
149 
150  switch (cmd) {
151  /* Make sure that no pending write process */
152  case CTRL_SYNC:
153  res = RES_OK;
154  break;
155 
156  /* Get number of sectors on the disk (DWORD) */
157  case GET_SECTOR_COUNT:
158  BSP_SD_GetCardInfo(&CardInfo);
159  *(DWORD *)buff = CardInfo.LogBlockNbr;
160  res = RES_OK;
161  break;
162 
163  /* Get R/W sector size (WORD) */
164  case GET_SECTOR_SIZE:
165  BSP_SD_GetCardInfo(&CardInfo);
166  *(WORD *)buff = CardInfo.LogBlockSize;
167  res = RES_OK;
168  break;
169 
170  /* Get erase block size in unit of sector (DWORD) */
171  case GET_BLOCK_SIZE:
172  BSP_SD_GetCardInfo(&CardInfo);
173  *(DWORD *)buff = CardInfo.LogBlockSize / SD_DEFAULT_BLOCK_SIZE;
174  res = RES_OK;
175  break;
